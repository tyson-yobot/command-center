        // Real-time updates simulation
        function updateMetrics() {
            // Simulate real-time data updates
            const confidenceBar = document.querySelector('.confidence-fill');
            const newWidth = Math.random() * 20 + 75; // 75-95%
            confidenceBar.style.width = newWidth + '%';
            
            // Update processing numbers
            const leadsPerHour = document.querySelector('.metric-value');
            const currentValue = parseInt(leadsPerHour.textContent);
            const variation = Math.floor(Math.random() * 6) - 3; // -3 to +3
            leadsPerHour.textContent = Math.max(40, currentValue + variation);
        }

        // Update every 5 seconds
        setInterval(updateMetrics, 5000);

        // Add click handlers for activity items

                    this.style.background = 'rgba(255, 255, 255, 0.03)';
