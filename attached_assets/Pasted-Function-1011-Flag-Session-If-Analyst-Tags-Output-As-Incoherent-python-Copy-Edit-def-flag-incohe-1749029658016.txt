def log_rating_discrepancy(script_id, ratings_list):
    """
    Detects if analysts disagree significantly on outcome score.
    """
    if max(ratings_list) - min(ratings_list) > 2:
        log_command_center_event("⚖️ Rating Conflict Detected", f"{script_id} → Ratings: {ratings_list}")
        return True
    return False
