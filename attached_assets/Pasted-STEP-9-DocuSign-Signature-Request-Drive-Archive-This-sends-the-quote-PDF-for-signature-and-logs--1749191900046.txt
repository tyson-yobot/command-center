import requests

DOCUSIGN_ACCESS_TOKEN = "YOUR_DOCUSIGN_TOKEN"
ACCOUNT_ID = "YOUR_DOCUSIGN_ACCOUNT_ID"
TEMPLATE_ID = "YOUR_TEMPLATE_ID"

def send_docusign_signature(client_email, client_name, pdf_path, company_name):
    url = f"https://demo.docusign.net/restapi/v2.1/accounts/{ACCOUNT_ID}/envelopes"

    headers = {
        "Authorization": f"Bearer {DOCUSIGN_ACCESS_TOKEN}",
        "Content-Type": "application/json"
    }

    with open(pdf_path, "rb") as file:
        file_bytes = file.read()

    data = {
        "emailSubject": f"Please sign the YoBot¬Æ Quote ‚Äì {company_name}",
        "documents": [{
            "documentBase64": file_bytes.encode("base64").decode(),
            "name": f"{company_name} Quote",
            "fileExtension": "pdf",
            "documentId": "1"
        }],
        "recipients": {
            "signers": [{
                "email": client_email,
                "name": client_name,
                "recipientId": "1",
                "routingOrder": "1",
                "tabs": {
                    "signHereTabs": [{
                        "anchorString": "/sig1/",
                        "anchorUnits": "pixels",
                        "anchorYOffset": "10",
                        "anchorXOffset": "20"
                    }]
                }
            }]
        },
        "status": "sent"
    }

    response = requests.post(url, headers=headers, json=data)
    if response.status_code == 201:
        envelope_id = response.json()['envelopeId']
        print(f"‚úÖ DocuSign sent: {envelope_id}")
        return envelope_id
    else:
        print(f"‚ùå DocuSign failed: {response.text}")
        return None
