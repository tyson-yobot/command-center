def normalize_speech_like_input(script_id, user_text):
    """
    Adjusts punctuation or spacing to simulate natural pacing.
    """
    if "," not in user_text and "." not in user_text:
        log_command_center_event("🗣️ Pacing Normalized", f"{script_id} → Input lacked pauses")
        return True
    return False
